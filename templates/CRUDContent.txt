component extends="Controller" { 

	function init() { 
		// Verify Params
		verifies(except="index,new,create", params="key", paramsTypes="integer", handler="objectNotFound");
	}

	/**
	*  @hint View all |ObjectNamePlural|
	*/
	function index() {
		|ObjectNamePlural|=model("|ObjectNameSingular|").findAll();
	}

	/**
	*  @hint View |ObjectNameSingular|
	*/
	function show() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").findByKey(params.key); 
	}

	/**
	*  @hint Add New |ObjectNameSingular|
	*/
	function new() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").new();
	}

	/**
	*  @hint Create |ObjectNameSingular|
	*/
	function create() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").create(params.|ObjectNameSingular|);
		if(|ObjectNameSingular|.hasErrors()){
			renderPage(action="new");
		} else {
			redirectTo(action="index", success="|ObjectNameSingular| successfully created"); 
		}		
	}

	/**
	*  @hint Edit |ObjectNameSingular|
	*/
	function edit() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").findByKey(params.key); 
	}

	/**
	*  @hint Update |ObjectNameSingular|
	*/
	function update() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").findByKey(params.key);
		if(|ObjectNameSingular|.update(params.|ObjectNameSingular|)){
			redirectTo(action="index", success="|ObjectNameSingular| successfully updated");
		} else {
			renderPage(action="edit");
		}
	}	

	/**
	*  @hint Delete |ObjectNameSingular|
	*/
	function delete() {
		|ObjectNameSingular|=model("|ObjectNameSingular|").deleteByKey(params.key);		
		redirectTo(action="index", success="|ObjectNameSingular| successfully deleted");
	} 	

	/**
	*  @hint Redirect away if verifies fails, or if an object can't be found
	*/
	function objectNotFound() { 
		redirectTo(action="index", error="That |ObjectNameSingular| wasn't found");
	} 
	
}
